<script>
    // --- SIDEBAR LOGIC ---
    function toggleSidebar() {
        const sidebar = document.getElementById("sidebar");
        const wrapper = document.getElementById("main-wrapper");
        const overlay = document.getElementById("sidebar-overlay");

        if (window.innerWidth < 992) {
            sidebar.classList.toggle("show");
            overlay.classList.toggle("show");
        } else {
            sidebar.classList.toggle("collapsed");
            wrapper.classList.toggle("full");
        }
    }

    // --- EXECUTE ON WINDOW LOAD (Visual Finish) ---
    window.addEventListener('load', function() {
        // Hide Rocket Loader
        const loader = document.getElementById('loader-wrapper');
        if (loader) {
            loader.style.opacity = '0';
            setTimeout(() => { loader.style.display = 'none'; }, 600);
        }

        // Finish Top Progress Bar
        const bar = document.getElementById('nprogress-bar');
        if (bar) {
            bar.style.width = '100%';
            setTimeout(() => { bar.style.opacity = '0'; }, 500);
        }
    });

    // --- EXECUTE ON DOM READY (Interactive Elements) ---
    $(document).ready(function() {
        
        // A. CSRF Setup for AJAX
        $.ajaxSetup({
            beforeSend: function(xhr, settings) {
                if (!/^(GET|HEAD|OPTIONS|TRACE)$/i.test(settings.type) && !this.crossDomain) {
                    xhr.setRequestHeader("X-CSRFToken", "{{ csrf_token }}");
                }
            }
        });

        // B. Initialize & Show Toasts
        $('.toast').each(function() {
            let t = new bootstrap.Toast(this, { delay: 5000 });
            t.show();
        });

        // C. Auto-Dismiss Top Alerts (The fix you needed)
        function dismissAlerts() {
            const alerts = document.querySelectorAll('.alert');
            alerts.forEach(function(alert) {
                setTimeout(function() {
                    let bsAlert = bootstrap.Alert.getOrCreateInstance(alert);
                    if (bsAlert) bsAlert.close();
                }, 5000);
            });
        }
        dismissAlerts();

        // D. Re-run dismiss for AJAX-loaded alerts (optional but pro)
        $(document).ajaxSuccess(function() {
            dismissAlerts();
        });
    });
</script>
